		public bool GetDateRange(DateTime curdt, ref DateTime startdate, ref DateTime endDate) {
			bool result = false;
			try {
				DateTime lastweek = curdt.AddDays(-7);
				int day = (int)lastweek.DayOfWeek;
				int diff = 0;
				if (day == 1) {
					diff = 0;
				}
				else if (day == 0) {
					diff = 1;
				}
				else {
					diff = 1 - day;
				}
				startdate = lastweek.AddDays(diff);
				endDate = startdate.AddDays(7);
				result = true;
			}
			catch (Exception) {
			}
			return result;
